#version 150

uniform mat4 proj;
uniform mat4 view;
uniform mat4 model;

in vec3 vsInPosition;
in vec2 vsInTexCoord;
in vec3 vsInColor;

out vec2 vsOutTexCoord;
out vec3 vsOutColor;

void main()
{
	// Pass the tex coord straight through to the fragment shader
	vsOutTexCoord = vsInTexCoord;
	vsOutColor = vsInColor;
	
	gl_Position = proj * view * model * vec4(vsInPosition, 1.0);
}
